public class ATMAccount {
    long accountNumber;
    double balance;
    
public static void withdraw(double amount, ATMAccount account){
    
        if(amount>account.balance){
            System.out.println("RS. "+amount+" rejected insuficient funds plese try again");
        }else if(amount<0){
        double remainingBalance = account.balance -(-amount);
            System.out.println("Withdraw succesful RS. "+amount+ " Remaining Balance RS. "+remainingBalance);
        }else {
            double updatedBalance = account.balance + amount;
            System.out.println("Deposite Succesful RS. "+amount+" Updated Balance RS. "+updatedBalance);
        }
    }
    
    public static void main(String []args){
        ATMAccount account = new ATMAccount();
        account.accountNumber = 12345678901L;
        account.balance = 10000.0;
        ATMAccount.withdraw(12000, account);
}
}
